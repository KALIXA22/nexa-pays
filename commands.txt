# ESP8266 MicroPython Setup Commands - Run in this order:

# Step 1: Erase flash memory
python -m esptool --port COM3 erase_flash

# Step 2: Flash MicroPython firmware
python -m esptool --port COM3 --baud 460800 write-flash --flash-size detect 0 "C:\Users\HHP\Downloads\ESP8266_GENERIC-20251209-v1.27.0.bin"

ampy --port COM3 put main.py

# Notes:
# - Make sure ESP8266 is connected to COM3 (check Device Manager)
# - Install required tools: pip install esptool ampy
# - Have mfrc522.py and main.py files ready in your current directory
